def store_dir
  if %w[development test].include?(Rails.env)
    "#{Rails.root.join('storage')}/#{model.class.to_s.underscore}/#{mounted_as}/#{model.id}"
  else
    "#{Rails.root.join('decidim_uploads')}/#{model.class.to_s.underscore}/#{mounted_as}/#{model.id}"
  end
end
