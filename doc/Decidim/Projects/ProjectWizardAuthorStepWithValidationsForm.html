def validate_zip_code
  return errors.add(:zip_code, 'nie może być puste') if zip_code.empty?
  return errors.add(:zip_code, 'jest nie poprawne') if zip_code.scan(/\A[0-9]{2}-[0-9]{3}\z/).empty?
