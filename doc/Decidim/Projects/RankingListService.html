def assign_status_to_project_ranks
  # clean statuses
  @project_ranks.within_minimum_votes_limit.update_all(status: nil)

  Decidim::Scope.all.each do |scope|
    calculate_status_for_scope(scope, @project_ranks.within_minimum_votes_limit.where(scope_id: scope.id))
  end
end

def calculate_status_for_scope(scope, collection)
  scope_value = @process.scope_budgets.find_by(decidim_scope_id: scope.id)&.budget_value
  return if scope_value.nil?

  sum = 0

  collection.order(valid_votes_count: :desc).each do |pr|
    summed = sum + pr.project.budget_value
    if summed <= scope_value
      sum = summed
      pr.update_column('status', 'on_the_list')
    else
      break
    end
  end
end

def projects_minimum(project)
  project.decidim_scope_id == Decidim::Projects::Project::GLOBAL_SCOPE_ID ? project.minimum_global_scope_projects_votes : project.minimum_district_scope_projects_votes
end

def verify_votes_count_treshold
  @project_ranks.each do |pr|
    pr.update_column('votes_limit_reached', votes_limit_reached?(pr.valid_votes_count, projects_minimum(pr.project)))
  end
  return ''
end
