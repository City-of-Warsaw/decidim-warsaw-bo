def call
  transaction do
    create_or_update_projects_ranks
          Decidim::Projects::ProjectRank.update_all(drawing_projects_log_id: nil)
    Decidim::Projects::DrawingProjectsLog.delete_all

    update_positions_on_ranking_lists
  end

  broadcast(:ok)
end

def project_rank_attributes(project)
  {
    project_id: project.id,
    scope_id: project.decidim_scope_id,
    valid_votes_count: project.votes.valid.count,
    not_verified_votes_count: project.votes.not_verified.count,
    invalid_votes_count: project.votes.invalid.count,
    votes_in_paper_count: project.votes.in_paper.count,
    votes_electronic_count: project.votes.electronic.count,
    percentage_of_not_verified_votes: percentage_of_not_verified_votes(project),
    votes_count: project.votes.in_paper.count + project.votes.electronic.count
  }
end

def update_positions_on_ranking_list(scope)
  ranks_collection = project_ranks.within_minimum_votes_limit.for_scope(scope).sorted_by_valid_votes_count
  max_budget_value = @edition.budget_value_for(scope)
  return if max_budget_value.nil?

  current_total = 0.00
  ranks_collection.group(:valid_votes_count).count.each do |votes_count, projects_number|
    # => { 2=>1, 1=>4, 0=>1 }
    next if votes_count.zero? # pomijamy projekty, ktore nie zdobyly glosow

    # jesli ilosc projektow z ta sama liczba glosow = 1
    if projects_number == 1
      pr = ranks_collection.find_by(valid_votes_count: votes_count)
    if current_total + pr.project.budget_value <= max_budget_value
        current_total += pr.project.budget_value
        pr.update_column('status', 'on_the_list')
      else
        # projekt nie miesci sie w budzecie, pomijamy
      end
    else
      # jest wiecej projektow z ta sama iloscia glosow,
      ranks = ranks_collection.where(valid_votes_count: votes_count)
      sum = ranks.map{ |r| r.project.budget_value }.sum
      if current_total + sum <= max_budget_value
        # wszystkie wpadaja na liste
        ranks.each do |pr|
          current_total += pr.project.budget_value
          pr.update_column('status', 'on_the_list')
        end
      else
        # losowanie - losujemy kolejnosc i dodajemy pojedynczo do listy poki mieszcza sie w budzecie
        drawing_projects_log = Decidim::Projects::DrawingProjectsLog.new
        drawing_projects_log.all_project_ids = ranks.map{ |r| r.project_id }
        shuffled_ranks = ranks.shuffle
        shuffled_ranks.each do |pr|
          if current_total + pr.project.budget_value <= max_budget_value
            current_total += pr.project.budget_value
            pr.update_column('status', 'on_the_list')
            drawing_projects_log.winning_project_ids << pr.project_id
          else
            # projekt nie miesci sie w budzecie, pomijamy
          end
        end
        drawing_projects_log.save
        # aktualizujemy informacje o losowaniu w projet_ranks
        shuffled_ranks.each{ |r| r.update_column('drawing_projects_log_id', drawing_projects_log.id) }
      end

    end
  end
end
