def calculate_percentage(not_verified_votes_count, submitted)

  (not_verified_votes_count / submitted.count) * 100
end

def call
  # return broadcast(:invalid) if form.invalid?

  transaction do
    projects.each do |project|
      if project.project_rank
        update_project_rank(project)
      else
        generate_project_rank(project)
      end
    end
    project_ranks = Decidim::Projects::ProjectRank.where(project_id: projects.pluck(:id))
    Decidim::Projects::RankingListService.new(project_ranks, @current_space).verify_votes_count_treshold
  end

  broadcast(:ok)
end

def project_rank_attributes(project)
  {
    project_id: project.id,
    scope_id: project.decidim_scope_id,
    valid_votes_count: project.votes.valid.count,
    not_verified_votes_count: project.votes.not_verified.count,
    invalid_votes_count: project.votes.invalid.count,
    votes_in_paper_count: project.votes.in_paper.count,
    votes_electronic_count: project.votes.electronic.count,
    percentage_of_not_verified_votes: calculate_percentage(project.votes.not_verified.count, project.votes.submitted)
  }
end
