def call
  return broadcast(:invalid) unless vote.status == Decidim::Projects::Vote::STATUSES::WAITING

  transaction do
    publish_vote
  end

  broadcast(:ok, @vote)
end

def publish_vote
  PaperTrail.request(enabled: false) do
    @vote = Decidim.traceability.perform_action!(
      :publish,
      @vote,
      @current_user,
      visibility: "admin-only"
    ) do
      @vote.update_columns(vote_attributes)
          @vote
    end
  end
end
