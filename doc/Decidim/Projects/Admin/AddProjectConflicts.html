def add_project_conflicts
  project.projects_in_conflict = form.conflicted_projects
  project_id = project.id
  # Adding mirrored association
  form.conflicted_projects.each do |proj|
    already_added_projects_ids = proj.projects_in_conflict.pluck(:id) + [project_id]
    proj.projects_in_conflict = Decidim::Projects::Project.where('decidim_projects_projects.id': already_added_projects_ids)
  end
end

def call
  return broadcast(:invalid) unless project
  return broadcast(:invalid) if form.invalid?

  transaction do
    add_project_conflicts
  end

  broadcast(:ok, project)
end
