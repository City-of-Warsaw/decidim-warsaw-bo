def query
  votes = Vote.where(component: component)
  return votes if user.ad_admin?

  if user.ad_coordinator? && user.assigned_scope_id
    # coordinator can see votes only when they need to be accepted
    votes.paper_votes.where('decidim_projects_votes.scope_id': user.assigned_scope_id).waiting_votes
  elsif user.ad_editor?
    # editor can see only votes he added and as long as they were not accepted
    votes.paper_votes.where(author_id: user.id).waiting_votes
  else
    # for coordinators from departments that are not assigned to scope
    Vote.none
  end
end
