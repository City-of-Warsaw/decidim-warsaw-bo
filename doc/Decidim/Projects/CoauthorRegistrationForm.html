def correct_password(pass)

  !!pass.match(/[a-z]/) && !!pass.match(/[A-Z]/) && !!pass.match(/[0-9]/) && !!pass.match(/[!@#$%^&*()_,<>';\\\]\[|":{}=\-\+\.?]/)
end
