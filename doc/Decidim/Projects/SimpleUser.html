def anonymous_name(with_number = false)
  an = with_number ? " #{anonymous_number}" : ''
  if gender.present?
    "#{I18n.t(gender, scope: 'decidim.users.gender', default: I18n.t('decidim.users.gender.male'))}#{an}"
  else
    "Mieszkaniec#{an}"
  end
end

def name(with_number = false)

    name_and_surname
  else
    an = with_number ? " #{anonymous_number}" : ''

      "Mieszkaniec#{an}"
    end
  end
