def call
  return broadcast(:invalid) if form.invalid?

  if vote_with_active_link
    send_email_notification(vote_with_active_link)
    return broadcast(:send_again)
  else
    transaction do
      create_votes_card
      send_email_notification(@votes_card)
    end
  end

  broadcast(:ok)
end

def generate_token
  token = SecureRandom.hex(36)
  while Decidim::Projects::Vote.find_by(voting_token: token)
    token = SecureRandom.hex(36)
  end
  token
end

def vote_attributes
  {
    status: Decidim::Projects::Vote::STATUSES::LINK_SENT,
    decidim_component_id: current_component.id,
    email: email,
    card_number: Decidim::Projects::Vote.generate_card_number(current_component),
    voting_token: generate_token,
    ip_number: client_ip,
    city: 'Warszawa' # default
  }
end

def vote_with_active_link
  @vote_with_active_link ||= Decidim::Projects::Vote.where(decidim_component_id: current_component.id).where(email: email).active_link.first
