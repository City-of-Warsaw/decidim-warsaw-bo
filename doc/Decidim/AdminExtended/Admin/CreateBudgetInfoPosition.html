def budget_info_position_attributes
  {
    name: @form.name,
    description: @form.description,
    amount: @form.amount,
    file: @form.file,
    published: @form.published,
    weight: @form.weight,
    budget_info_group_id: @form.budget_info_group_id,
    on_main_site: @form.on_main_site
  }
end

  create_budget_info_position
  broadcast(:ok)
end
